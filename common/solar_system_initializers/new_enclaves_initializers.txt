# Spiritualists
ne_shroud_init = {
	class = sc_black_hole
	usage = misc_system_init

	usage_odds = {
		base = 100
		modifier = {
			factor = 0
			is_fe_cluster = yes
		}
		modifier = {
			factor = 0
			has_star_flag = empire_cluster
		}
	}

	max_instances = 1
	spawn_chance = 100

	flags = { ne_shroud_system ne_enclave }

	planet = {
		name = "NAME_NESHR_Star"
		class = star
		orbit_distance = 0
		orbit_angle = 1
		size = 30
		has_ring = no


		init_effect = {
			if = {
				limit = {
					NOT = { exists = event_target:ne_shroud_country }
				}
				create_species = {
					name = random
					class = random_non_machine
					portrait = random
					traits = {
						trait = trait_psionic
						trait = trait_venerable
					}
				}
				create_country = {
					name = "NAME_NESHR_Country"
					type = ne_enclave
					authority = "auth_oligarchic"
					civics =	{
						civic = civic_priesthood_of_the_cycle
					}
					species = last_created
					flag = {
						icon = {
							category = "enclaves"
							file = "ne_flag_shroud.dds"
						}
						background= {
							category = "backgrounds"
							file = "stripes.dds"
						}
						colors={
							"purple"
							"dark_blue"
							"null"
							"null"
						}
					}
					ethos = {
						ethic = ethic_fanatic_spiritualist
						ethic = ethic_pacifist
					}
					ignore_initial_colony_error = yes
				}
				last_created_country = {
					set_country_flag = ne_shroud_country
					set_graphical_culture = mammalian_01
					save_global_event_target_as = ne_shroud_country
					create_fleet = {
						name = "NAME_NESHR_Fleet"
						settings = { spawn_debris = no }
						effect = {
							set_owner = prev
								create_ship = {
								name = random
								design = "NAME_NESHR_Enclave_Station"
								graphical_culture = prev
							}
							set_location = {
								target = prevprev
								angle = random
								distance = 100
							}
							save_event_target_as = CustomCapital
							prev = {
								set_custom_capital_location = event_target:CustomCapital
							}
						}
					}
					create_leader = {
						class = ruler
						species = prev
						name = random
						skill = 5
					}
					set_leader = last_created_country
				}
			}
		}
	}

	planet = {
		name = "NAME_NESHR_Planet_1"
		class = "pc_shrouded"
		orbit_distance = 40
		orbit_angle = 90
		size = 25
		has_ring = no
	}

	planet = {
		name = "NAME_NESHR_Planet_2"
		class = "pc_shielded"
		orbit_distance = 40
		orbit_angle = 2
		size = 20
		has_ring = no
	}

	planet = {
		name = "NAME_NESHR_Planet_3"
		class = "pc_broken"
		orbit_distance = 40
		orbit_angle = 2
		size = 15
		has_ring = no
	}
}
