              category = "backgrounds"

            }

        last_created_country = {
          set_country_flag = ne_spiritualist_country
          set_graphical_culture = mammalian_01
          save_global_event_target_as = ne_spiritualist_country
          create_fleet = {
            name = "NAME_Great_Temple"
            settings = { spawn_debris = no }

              set_owner_prev
              create_ship = {
                name = random
                design = "NAME_NE_Spiritutalist_Station"

                target = prevprev

                distance = 100

                set_custom_capital_location = event_target:CustomCapital

            }
          }

            class = ruler
            species = prev
            name = random
            skill = 5
          }
          set_leader = last_created_country

    }
